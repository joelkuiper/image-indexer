# Agentic CLI Implementation Plan

> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task.

**Goal:** Build a CLI that humans and AI agents both love — `idx` command with JSON-first output, clear exit codes, and zero interactive prompts.

**Architecture:** Click-based CLI with three commands (`index`, `search`, `status`) that wrap existing preprocessing, RunPod client, and database layers. All commands support `--json` flag for machine consumption and default to human-readable tables.

**Tech Stack:** Click 8.x (already in deps), Rich for tables (optional, human-mode only), existing `image_indexer.preprocess`, `image_indexer.db`, and new `image_indexer.client` module.

---

## Design Principles

1. **JSON-first:** Every command returns structured data. `--json` flag (default: off for humans, but agents always use it).
2. **Exit codes tell the story:** 0 = success, 1 = user error (bad args), 2 = system error (DB/network), 3 = partial failure (some images failed).
3. **Non-interactive:** No prompts, no confirmations. Fail fast with clear errors.
4. **Logging is separate:** `--verbose` shows progress to stderr, data goes to stdout. Agents parse stdout, ignore stderr.
5. **Idempotent:** Running `idx index <dir>` twice indexes new images, updates changed ones, skips unchanged (SHA-256 dedup).

---

## Command Structure

```bash
# Index a directory
idx index ~/Photos/2024 --json --verbose
# Output: {"indexed": 142, "skipped": 3, "failed": 0, "database": "/home/joel/.local/share/image-indexer/db.sqlite"}
# Exit: 0 (success) or 3 (partial failure if some failed)

# Search images
idx search "sunset beach" --semantic --json
# Output: [{"path": "...", "score": 0.87, "description": "..."}, ...]
# Exit: 0

# Show database status
idx status --json
# Output: {"images": 142, "database_path": "...", "last_indexed": "2026-06-14T10:30:00Z"}
# Exit: 0
```

"""Agentic CLI for image indexing.

Commands:
  index   - Index a directory of images
  search  - Search indexed images
  status  - Show database stats
"""
import sys
import json
import click
from pathlib import Path

DEFAULT_DB = Path.home() / ".local" / "share" / "image-indexer" / "db.sqlite"

@click.group()
@click.option("--json", "output_json", is_flag=True, help="Output JSON instead of tables")
@click.option("--verbose", is_flag=True, help="Show progress to stderr")
@click.option("--db", type=click.Path(), default=str(DEFAULT_DB), help="Database path")
@click.pass_context
def cli(ctx, output_json, verbose, db):
    """Image indexer CLI.

    Agentic-friendly: --json for machines, tables for humans.
    Exit codes: 0=success, 1=user error, 2=system error, 3=partial failure.
    """
    ctx.ensure_object(dict)
    ctx.obj["json"] = output_json
    ctx.obj["verbose"] = verbose
    ctx.obj["db"] = Path(db)

if __name__ == "__main__":
    cli()
```

@cli.command()
@click.pass_context
def status(ctx):
    """Show database statistics."""
    from image_indexer.db import connect
    
    try:
        db = connect(ctx.obj["db"])
    except Exception as e:
        click.echo(f"Error connecting to database: {e}", err=True)
        sys.exit(2)
    
    # Query stats
    image_count = db.execute("SELECT COUNT(*) FROM images").fetchone()[0]
    
    # Get last indexed timestamp (if any)
    last_row = db.execute(
        "SELECT MAX(updated_at) FROM images"
    ).fetchone()
    last_indexed = last_row[0] if last_row[0] else None
    
    result = {
        "database_path": str(ctx.obj["db"]),
        "images": image_count,
        "last_indexed": last_indexed,
    }
    
    if ctx.obj["json"]:
        click.echo(json.dumps(result, indent=2))
    else:
        # Human-readable table
        click.echo(f"Database: {result['database_path']}")
        click.echo(f"Images:   {result['images']}")
        if result['last_indexed']:
            click.echo(f"Last indexed: {result['last_indexed']}")
    
    sys.exit(0)
```

```python
@cli.command()
@click.argument("query")
@click.option("--semantic", is_flag=True, help="Semantic search (vector similarity)")
@click.option("--lexical", is_flag=True, help="Lexical search (FTS5 full-text)")
@click.option("--structured", is_flag=True, help="Structured search (SQL WHERE clause)")
@click.option("--limit", type=int, default=10, help="Max results (default: 10)")
@click.pass_context
def search(ctx, query, semantic, lexical, structured, limit):
    """Search indexed images.

    QUERY is the search term or SQL WHERE clause.

    Examples:
      idx search "sunset beach" --semantic
      idx search "beach" --lexical
      idx search "camera_model = 'iPhone'" --structured
    """
    from image_indexer.db import connect
    
    # Validate: at least one mode must be specified
    if not (semantic or lexical or structured):
        click.echo("Error: specify at least one of --semantic, --lexical, --structured", err=True)
        sys.exit(1)
    
    try:
        db = connect(ctx.obj["db"])
    except Exception as e:
        click.echo(f"Error connecting to database: {e}", err=True)
        sys.exit(2)
    
    results = []
    
    if semantic:
        # Semantic search requires a text embedding from RunPod
        # For now, return error (requires RunPod endpoint)
        click.echo("Semantic search not yet implemented (requires RunPod endpoint)", err=True)
        sys.exit(2)
    
    if lexical:
        from image_indexer.db import search_lexical
        results = search_lexical(db, query, k=limit)
    
    if structured:
        from image_indexer.db import search_structured
        # Structured query is the WHERE clause itself
        results = search_structured(db, query)
    
    if ctx.obj["json"]:
        click.echo(json.dumps(results, indent=2, default=str))
    else:
        # Human-readable table
        if not results:
            click.echo("No results found.")
        else:
            click.echo(f"Found {len(results)} results:")
            for row in results:
                click.echo(f"  {row.get('path', 'N/A')}")
                if 'description' in row:
                    desc = row['description'][:80] + "..." if len(row['description']) > 80 else row['description']
                    click.echo(f"    {desc}")
    
    sys.exit(0)
```

**Step 2: Implement RunPod client**

```python
"""RunPod HTTP client for image processing."""
import base64
import time
import requests

class RunPodClient:
    """Client for RunPod serverless endpoints."""
    
    def __init__(self, api_key: str, endpoint_id: str, timeout: int = 300):
        self.api_key = api_key
        self.endpoint_id = endpoint_id
        self.timeout = timeout
        self.base_url = f"https://api.runpod.ai/v2/{endpoint_id}"
    
    def process_image(self, image_bytes: bytes, task: str = "all") -> dict:
        """Send image to RunPod and return embedding + description.
        
        Args:
            image_bytes: Raw image bytes (JPEG/PNG)
            task: "embed", "caption", or "all"
        
        Returns:
            dict with "embedding" (list[float]) and "description" (str)
        
        Raises:
            RuntimeError: If RunPod returns an error
        """
        # Encode image
        image_b64 = base64.b64encode(image_bytes).decode("utf-8")
        
        # Submit job
        response = requests.post(
            f"{self.base_url}/run",
            headers={
                "Content-Type": "application/json",
                "Authorization": f"Bearer ******",
            },
            json={"input": {"image_b64": image_b64, "task": task}},
            timeout=self.timeout,
        )
        
        if response.status_code != 200:
            raise RuntimeError(f"RunPod HTTP {response.status_code}: {response.text}")
        
        result = response.json()
        
        # Check for errors
        if "error" in result:
            raise RuntimeError(f"RunPod error: {result['error']}")
        
        # Poll for completion (if async)
        if result.get("status") == "IN_QUEUE" or result.get("status") == "IN_PROGRESS":
            result = self._poll_until_complete(result["id"])
        
        # Extract output
        if result.get("status") != "COMPLETED":
            raise RuntimeError(f"Unexpected status: {result.get('status')}")
        
        return result.get("output", {})
    
    def _poll_until_complete(self, request_id: str, poll_interval: float = 2.0) -> dict:
        """Poll RunPod status endpoint until job completes."""
        start_time = time.time()
        
        while time.time() - start_time < self.timeout:
            response = requests.get(
                f"{self.base_url}/status/{request_id}",
                headers={"Authorization": f"Bearer ******"},
                timeout=30,
            )
            
            if response.status_code != 200:
                raise RuntimeError(f"Status check failed: HTTP {response.status_code}")
            
            result = response.json()
            status = result.get("status")
            
            if status == "COMPLETED":
                return result
            elif status in ("FAILED", "CANCELLED"):
                raise RuntimeError(f"Job {status}: {result.get('error', 'Unknown error')}")
            
            time.sleep(poll_interval)
        
        raise RuntimeError(f"Timeout waiting for job {request_id}")
```

**Step 5: Implement index command**

```python
@cli.command()
@click.argument("directory", type=click.Path(exists=True, file_okay=False, resolve_path=True))
@click.option("--endpoint-id", envvar="RUNPOD_ENDPOINT_ID", help="RunPod endpoint ID (or set RUNPOD_ENDPOINT_ID)")
@click.option("--api-key", envvar="RUNPOD_API_KEY", help="RunPod API key (or set RUNPOD_API_KEY)")
@click.option("--dry-run", is_flag=True, help="Preprocess but don't send to RunPod")
@click.pass_context
def index(ctx, directory, endpoint_id, api_key, dry_run):
    """Index a directory of images.

    Process images: preprocess → RunPod inference → store in database.

    Examples:
      idx index ~/Photos/2024
      idx index ./scans --dry-run  # Preprocess only, no RunPod
    """
    from image_indexer.preprocess import preprocess, scan_directory
    from image_indexer.db import connect, upsert_image
    
    # Validate RunPod credentials (unless dry-run)
    if not dry_run:
        if not endpoint_id:
            click.echo("Error: --endpoint-id or RUNPOD_ENDPOINT_ID required", err=True)
            sys.exit(1)
        if not api_key:
            click.echo("Error: --api-key or RUNPOD_API_KEY required", err=True)
            sys.exit(1)
    
    # Scan directory
    directory = Path(directory)
    if ctx.obj["verbose"]:
        click.echo(f"Scanning {directory}...", err=True)
    
    image_paths = scan_directory(directory)
    
    if ctx.obj["verbose"]:
        click.echo(f"Found {len(image_paths)} images", err=True)
    
    # Connect to database
    try:
        db = connect(ctx.obj["db"])
    except Exception as e:
        click.echo(f"Error connecting to database: {e}", err=True)
        sys.exit(2)
    
    # Initialize RunPod client (unless dry-run)
    client = None
    if not dry_run:
        from image_indexer.client import RunPodClient
        client = RunPodClient(api_key=api_key, endpoint_id=endpoint_id)
    
    # Process images
    indexed = 0
    skipped = 0
    failed = 0
    
    for path in image_paths:
        if ctx.obj["verbose"]:
            click.echo(f"  Processing {path.name}...", err=True)
        
        # Preprocess
        prep = preprocess(path)
        
        if prep.skipped:
            if ctx.obj["verbose"]:
                click.echo(f"    Skipped: {prep.skip_reason}", err=True)
            skipped += 1
            continue
        
        if dry_run:
            # Dry run: just preprocess, don't send to RunPod
            indexed += 1
            continue
        
        # Send to RunPod
        try:
            result = client.process_image(prep.jpeg_bytes, task="all")
            embedding = result.get("embedding")
            description = result.get("description")
        except Exception as e:
            if ctx.obj["verbose"]:
                click.echo(f"    Failed: {e}", err=True)
            failed += 1
            continue
        
        # Store in database
        try:
            meta = {
                "path": str(prep.path),
                "sha256": prep.sha256,
                "file_size": prep.file_size,
                "format": prep.disk_format,
                "width": prep.orig_width,
                "height": prep.orig_height,
                "description": description,
                "model_caption": "Qwen/Qwen3-VL-4B-Instruct",
                "model_embed": "google/siglip2-so400m-patch16-384",
            }
            upsert_image(db, meta, embedding=embedding)
            indexed += 1
        except Exception as e:
            if ctx.obj["verbose"]:
                click.echo(f"    DB error: {e}", err=True)
            failed += 1
            continue
    
    # Output results
    result = {
        "indexed": indexed,
        "skipped": skipped,
        "failed": failed,
        "database": str(ctx.obj["db"]),
    }
    
    if ctx.obj["json"]:
        click.echo(json.dumps(result, indent=2))
    else:
        click.echo(f"Indexed:  {indexed}")
        click.echo(f"Skipped:  {skipped}")
        click.echo(f"Failed:   {failed}")
        click.echo(f"Database: {ctx.obj['db']}")
    
    # Exit code: 3 if there were failures, 0 otherwise
    sys.exit(3 if failed > 0 else 0)
```

## Exit Codes

- `0` — Success
- `1` — User error (bad arguments, missing required options)
- `2` — System error (database connection, network failure)
- `3` — Partial failure (some images failed to index)

## Agent Usage

Agents should always use `--json` and check exit codes:

```bash
# Index and check for failures
result=$(idx index ~/Photos --json)
exit_code=$?

if [ $exit_code -eq 0 ]; then
    echo "Success"
elif [ $exit_code -eq 3 ]; then
    echo "Partial failure - check logs"
    echo "$result" | jq '.failed'
else
    echo "Error: exit code $exit_code"
    exit 1
fi
```

## Global Options

- `--json` — Output JSON instead of tables
- `--verbose` — Show progress to stderr
- `--db PATH` — Database path (default: `~/.local/share/image-indexer/db.sqlite`)
